Detailed Description

Functions for rendering system.

Example:

NRHandle rendering_handle = NRHandleNull;
NRResult result = NRRenderingCreate(&rendering_handle);
// Handle result.
uint32_t format_count = 0;
result = NRRenderingGetFormatsCount(rendering_handle, &format_count);
NRTextureFormat* formats = new NRTextureFormat[format_count];
result = NRRenderingGetFormats(rendering_handle, format_count, formats);
NRApiRequirements api_requirements;
result = NRRenderingGetApiRequirements(rendering_handle, &api_requirements);
uint32_t max_layer_count = 0;
result = NRRenderingGetMaxLayerCount(rendering_handle, &max_layer_count);
// Handle result.
result = NRRenderingStart(rendering_handle);
// Handle result.
NRHandle display_buffer_spec_handle = NRHandleNull;
result = NRBufferSpecCreate(rendering_handle, &display_buffer_spec_handle);
result = NRBufferSpecSetSize(rendering_handle, display_buffer_spec_handle, display_buffer_size);
uint32_t create_flags = NR_SWAPCHAIN_CREATE_FLAGS_NONE;
result = NRBufferSpecSetCreateFlags(rendering_handle, display_buffer_spec_handle, create_flags);
// Set the other buffer spec attributres
// Handle result.
// Create left display swap chain 0: rendering by left camera, and full screen display.
// Use NRSwapchainCreate to create the swap chain and native buffers.
// No need to call NRSwapchainSetBuffers anymore.
uint32_t left_swapchain_handle = NRHandleNull;
result = NRSwapchainCreate(rendering_handle, display_buffer_spec_handle, &left_swapchain_handle);
// Handle result.
// Create right display layer 1: rendering by right camera, and full screen display.
// Use NRSwapchainCreateEx to create the swap chain without native buffers.
// Need to call NRSwapchainSetBuffers later.
uint32_t right_swapchain_handle = NRHandleNull;
result = NRSwapchainCreateEx(rendering_handle, display_buffer_spec_handle, &right_swapchain_handle);
result = NRSwapchainGetRecommendBufferCount(rendering_handle, right_swapchain_handle, &display_buffer_count);
// Handle result.
// New right display texture array, size is display_buffer_count.
result = NRSwapchainSetBuffers(rendering_handle, right_swapchain_handle, display_buffer_count, right_color_buffers);
// Handle result.
uint32_t video_buffer_flag = NR_SWAPCHAIN_EXTERNAL_SURFACE;
uint32_t video_buffer_count = 0;
result = NRRenderingGetRecommendBufferCount(rendering_handle, video_buffer_flag, video_buffer_count);
// Handle result.
NRHandle video_buffer_spec_handle = NRHandleNull;
result = NRBufferSpecCreate(rendering_handle, &video_buffer_spec_handle);
// Handle result.
result = NRBufferSpecSetSize(rendering_handle, video_buffer_spec_handle, video_buffer_size);
// Handle result.
// Create 2d video layer 3: read buffer from android surface.
uint32_t video_swapchain_handle = 0;
void* surface = nullptr;
result = NRSwapchainCreateAndroidSurface(rendering_handle, video_buffer_spec_handle, &video_swapchain_handle, &surface);
// Handle result. And set the surface as an output destination for the android.hardware.camera2, MediaCodec, MediaPlayer, and Allocation APIs.
NRHandle frame_handle = NRHandleNull;
while(true) {
result = NRRenderingAcquireFrame(rendering_handle, &frame_handle);
// Handle result.
uint32_t updated_swapchains[2] = {left_swapchain_handle, right_swapchain_handle};
NRTextureID updated_buffers[2];
result = NRFrameAcquireBuffers(rendering_handle, frame_handle, updated_swapchains, 2, updated_buffers);
// Handle result.
//
// wait vsync
//
uint64_t display_time = 0;
uint64_t dispaly_period = 0;
result = NRRenderingFrameWait(rendering_handle, &display_time, &dispaly_period);
// Handle result.
// get head pose using NRHeadTrackingAcquireHeadPose and set variable hmd_time_nanos to be display_time.
//
NRMat4f left_display_pose_from_head;
result = NRHMDGetComponentPoseFromHead(hmd_handle, NR_COMPONENT_DISPLAY_LEFT, &left_display_pose_from_head);
NRTransform left_camera_pose = head_pose * left_display_pose_from_head;
Use left_camera_pose to do left camera rendering, save the result in updated_color_buffers[0];
NRMat4f right_display_pose_from_head;
result = NRHMDGetComponentPoseFromHead(hmd_handle, NR_COMPONENT_DISPLAY_RIGHT, &right_display_pose_from_head);
NRTransform right_camera_pose = head_pose * right_display_pose_from_head;
Use right_camera_pose to do right camera rendering, save the result in updated_color_buffers[1];
result = NRFrameSetPresentTime(rendering_handle, frame_handle, display_time);
// Handle result.
NRHandle viewport_handle = NRHandleNull;
result = NRBufferViewportCreate(rendering_handle, &viewport_handle);
result = NRBufferViewportSetSourceUV(rendering_handle, viewport_handle, &left_video_uv_rect);
result = NRBufferViewportSetTargetComponent(rendering_handle, viewport_handle, NR_COMPONENT_DISPLAY_LEFT);
result = NRBufferViewportSetSwapchain(rendering_handle, viewport_handle, video_swapchain_handle);
result = NRBufferViewportSetType(rendering_handle, viewport_handle, NR_VIEWPORT_QUAD);
result = NRBufferViewportSetTransform(rendering_handle, viewport_handle, NR_REFERENCE_SPACE_GLOBAL, video_left_transform);
result = NRBufferViewportSetQuadSize(rendering_handle, viewport_handle, quad_width, quad_height);
result = NRFrameSetBufferViewport(rendering_handle, frame_handle, 0, viewport_handle);
result = NRBufferViewportSetSourceUV(rendering_handle, viewport_handle, &right_video_uv_rect);
result = NRBufferViewportSetTargetComponent(rendering_handle, viewport_handle, NR_COMPONENT_DISPLAY_RIGHT);
result = NRBufferViewportSetSwapchain(rendering_handle, viewport_handle, video_swapchain_handle);
result = NRBufferViewportSetType(rendering_handle, viewport_handle, NR_VIEWPORT_QUAD);
result = NRBufferViewportSetTransform(rendering_handle, viewport_handle, NR_REFERENCE_SPACE_GLOBAL, video_right_transform);
result = NRBufferViewportSetQuadSize(rendering_handle, viewport_handle, quad_width, quad_height);
result = NRFrameSetBufferViewport(rendering_handle, frame_handle, 1, viewport_handle);
result = NRBufferViewportSetSourceUV(rendering_handle, viewport_handle, &left_display_uv_rect);
result = NRBufferViewportSetTargetComponent(rendering_handle, viewport_handle, NR_COMPONENT_DISPLAY_LEFT);
result = NRBufferViewportSetSwapchain(rendering_handle, viewport_handle, left_swapchain_handle);
result = NRBufferViewportSetType(rendering_handle, viewport_handle, NR_VIEWPORT_PROJECTION);
result = NRBufferViewportSetSourceFov(rendering_handle, viewport_handle, left_camera_fov);
result = NRBufferViewportSetTransform(rendering_handle, viewport_handle, NR_REFERENCE_SPACE_GLOBAL, left_camera_pose);
result = NRBufferViewportSetFocusPlane(rendering_handle, viewport_handle, NR_REFERENCE_SPACE_GLOBAL, plane_in_global);
result = NRFrameSetBufferViewport(rendering_handle, frame_handle, 2, viewport_handle);
result = NRBufferViewportSetSourceUV(rendering_handle, viewport_handle, &right_display_uv_rect);
result = NRBufferViewportSetTargetComponent(rendering_handle, viewport_handle, NR_COMPONENT_DISPLAY_RIGHT);
result = NRBufferViewportSetSwapchain(rendering_handle, viewport_handle, right_swapchain_handle);
result = NRBufferViewportSetType(rendering_handle, viewport_handle, NR_VIEWPORT_PROJECTION);
result = NRBufferViewportSetSourceFov(rendering_handle, viewport_handle, right_camera_fov);
result = NRBufferViewportSetTransform(rendering_handle, viewport_handle, NR_REFERENCE_SPACE_GLOBAL, right_camera_pose);
result = NRBufferViewportSetFocusPlane(rendering_handle, viewport_handle, NR_REFERENCE_SPACE_GLOBAL, plane_in_global);
result = NRFrameSetBufferViewport(rendering_handle, frame_handle, 3, viewport_handle);
result = NRFrameSubmit(rendering_handle, frame_handle);
result = NRBufferViewportDestroy(rendering_handle, viewport_handle);
// Handle result.
#endif
}
result = NRFrameDestroy(rendering_handle, frame_handle);
result = NRBufferSpecDestroy(rendering_handle, display_buffer_spec_handle);
result = NRBufferSpecDestroy(rendering_handle, video_buffer_spec_handle);
result = NRSwapchainDestroy(rendering_handle, swapchain_handle);
result = NRRenderingStop(rendering_handle);
// Handle result.
result = NRRenderingDestroy(rendering_handle);
// Handle result.

◆ NRRenderingAcquireFrame()

NRResult NRRenderingAcquireFrame ( NRHandle  rendering_handle,
NRHandle *  out_frame_handle 
)

Acquires the current frame from swap chain. Then users will update the buffers of this frame.

One frame contains several layers, one layer contains several buffers.

Calling this function will get the control of this frame. Calling NRFrameSubmit to submit the frame and return the control of frame back to SDK.

Parameters
[in]rendering_handleThe handle of rendering object.
[in]swapchain_handleThe handle of swap chian object.
[out]out_frame_handleThe handle of current frame whoes contents need to update.
Returns
The result of operation.

◆ NRRenderingDoRenderEx()

NRResult NRRenderingDoRenderEx ( NRHandle  rendering_handle,
NRHandle  frame_handle 
)

Submit frame to SDK rendering system.

Developer can render the left texture and right texture image to OpenglES GL_TEXTURE_2D or GL_TEXTURE_2D_ARRAY object. And SDK rendering system will do reprojection on these textures and present to display.

Note: Developers can convert from the GLuint to void* by type casting to get left_display_texture and right_display_texture parameters.

Parameters
[in]rendering_handleThe handle of rendering object.
[in]frame_handleThe handle of current frame.
Returns
The result of operation.
Deprecated:
Use NRFrameSubmit(...) instead.
