Detailed Description

Functions for managing viewer information.

Example:

NRHandle hmd_handle = NRHandleNull;
NRResult result = NRHMDCreate(&hmd_handle);
// Handle result.
NRMat4f component_pose_from_head;
result = NRHMDGetComponentPoseFromHead(hmd_handle, NRComponent::NR_COMPONENT_DISPLAY_LEFT,
&component_pose_from_head);
//
// Handle result and use component_pose_from_head;
//
NRFov4f component_fov;
result = NRHMDGetComponentFov(hmd_handle, NRComponent::NR_COMPONENT_DISPLAY_RIGHT, &component_fov);
//
// Handle result and use component_fov;
//
NRSize2i component_resolution;
result = NRHMDGetComponentResolution(hmd_handle, NRComponent::NR_COMPONENT_RGB_CAMERA,
&component_resolution);
//
// Handle result and use component_resolution;
//
result = NRHMDDestroy(hmd_handle);
// Handle result.
