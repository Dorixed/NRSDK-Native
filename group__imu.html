Detailed Description

Functions for managing viewer information.

Example:

//callback function
void ImuDataCallback(NRHandle imu_handle, NRHandle imu_data_handle, void*
usr_data) {
NRVector3f gyroscope;
NRVector3f accelerometer;
uint64_t hmd_time_nanos;
// Handle usr_data
NRResult result = NRImuDataGetGyroscope(imu_handle,
imu_data_handle,&gyroscope);
// Handle gyroscope
NRResult result = NRImuDataGetHMDTimeNanos(imu_handle,
imu_data_handle,&hmd_time_nanos);
// Handle hmd_time_nanos
result = NRImuDataGetAccelerometer(imu_handle,
imu_data_handle,&accelerometer);
// Handle accelerometer
NRImuDataDestroy(imu_handle, imu_data_handle);
}
//main func
NRHandle imu_handle;
NRResult result = NRImuCreate(&imu_handle);
// Handle result.
// set callback;
void* user_data;
result = NRImuSetCaptureCallback(imu_handle, ImuDataCallback,user_data);
// start
result = NRImuStart(imu_handle);
result = NRImuStop(imu_handle);
// Handle result.
result = NRImuDestroy(imu_handle);
// Handle result.
